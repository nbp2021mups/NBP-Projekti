// Nodes

User: {							[Korisnik]
	id: Int,					[Identifikator]
	username: String, 			[Korisničko ime]
	email: String,				[Email adresa]
	password: String,			[Lozinka]
	firstName: String,			[Ime]
	lastName: String,			[Prezime]
	image: String,				[Fotografija]
	bio: String,				[Opis]
	joined: Date,				[Korisnik od]
	friendsNo: Int				[Prijatelji]
	followedLocationsNo: Int	[Praćene lokacije]
	postsNo: Int				[Objave]
}

Location: {						[Lokacija]
	id: Int,					[Identifikator]
	city: String,				[Grad]
	country: String,			[Država]
	followersNo: Int,			[Pratioci]
	postsNo: Int,				[Objave]
	rating: Float				[Ocena]
}

Post: {							[Objava]
	id: Int,					[Identifikator]
	image: String,				[Fotografija]
	description: String,		[Opis]
	likesNo: Int,				[Broj lajkova]
	commentsNo: Int				[Broj komenatara]
}
//

// Relations

User -> Post
SHARED: {						[Objavljeno]
	id: Int,					[Identifikator]
	time: Date					[U]
}

User -> Post
LIKED: {						[Lajkovano]
	id: Int,					[Identifikator]
	time: Date					[U]
}

User -> Post
COMMENTED: {					[Komentarisano]
	id: Int,					[Identifikator]
	time: Date,					[U]
	comment: String				[Sadržaj]
}

Post -> Location
LOCATED_AT : {					[Na]
	id: Int						[Identifikator]
}

User -> Location
FOLLOWS: {						[Prati]
	id: Int,					[Identifikator]
	time: Date					[Od]
	notifications: Bool			[Omogućena obaveštenja]
}

User -> User
SENT_REQUEST: {
	id: Int						[Identifikator],
	time: Date					[Vreme]
}

User <-> User
IS_FRIEND: {					[Prijatelj]
	id: Int,					[Identifikator]
	chatId: Int,				[Identifikator konverzacije]
	since: Date					[Prijatelji od]
}
//



// Ovo će najverovatnije da ide u redis, ali ovo su atributi
Message: {						[Poruka]
	id: Int, 					[Identifikator]
	from: String,				[Od]
	to: String,					[Za]
	contentType: String,		[Vrsta sadržaja]
	content: String,			[Sadržaj]
	timeSent: Date,				[Poslato]
	timeRead: Date				[Viđeno]
}

Chat: {							[Ćaskanje]
	id: Int,					[Identifikator]
	user1Id: Int,				[Identifikator korisnika 1]
	user2Id: Int,				[Identifikator korisnika 2]
	messages ??					[Poruke]
}

Notification: {					[Obaveštenje]
	id: Int, 					[Identifikator]
	to: String,					[Za]
	type: String,				[Tip]
	content: String,			[Sadržaj]
	timeSent: Date,				[Poslato]
	read: bool					[Viđeno]
}
